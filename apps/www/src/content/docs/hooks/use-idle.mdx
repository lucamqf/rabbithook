Tracks the idle state of an user.

### Installation
<Code code="npx rabbithook@latest add use-idle" lang="bash" frame={false} />

### Usage
<Code
    lang="tsx"
    title="App.tsx"
    frame="code"
    code={`
import useIdle from "@/hooks/use-idle";

function Component() {
  const isIdle = useIdle(15 * 1000 /* 15 seconds */);

  return (
    <div>isIdle: { isIdle ? "Yes" : "No" }</div>
  )
}
`}
  />
