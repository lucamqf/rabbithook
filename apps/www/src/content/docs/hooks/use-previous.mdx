Tracks the changes of a given state and returns always its previous value.

### Installation
<Code code="npx rabbithook@latest add use-previous" lang="bash" frame={false} />

### Usage
<Code
    lang="tsx"
    title="App.tsx"
    frame="code"
    code={`
import usePrevious from "@/hooks/use-previous";

function Component() {
  const [count, setCount] = useState(0);
  const previousCount = usePrevious(count);

  return (
    <>
      <p>Count: { count }</p>                   {/* 0, 1, 2, 3, ... */}
      <p>Previous Count: { previousCount }</p>  {/* undefined, 0, 1, 2, ... */}
      <button onClick={() => setCount(prev => prev++)}>
        Increment
      </button>
    </>
  );
}
`}
  />
