Easy way of dealing with geolocation.

### Installation
<Code code="npx rabbithook@latest add use-geolocation" lang="bash" frame={false} />

### Usage
  <Code
    lang="tsx"
    title="App.tsx"
    frame="code"
    code={`
import useGeolocation from "@/hooks/use-geolocation";

function Component() {
  const { loading, error, data } = useGeolocation();

  if (loading) {
    return <p>Loading...</p>
  }

  if (error) {
    return <p>Error: {error.message}</p>
  }

  return (
    <p>{JSON.stringify(data)}</p>
  )
}
`}
  />
