Executes a function when the user long press a given element.

### Installation
<Code code="npx rabbithook@latest add use-long-press" lang="bash" frame={false} />

### Usage
<Code
    lang="tsx"
    title="App.tsx"
    frame="code"
    code={`
import useLongPress from "@/hooks/use-long-press";

function Component() {
  const ref = useRef<HTMLDivElement>(null);

  useLongPress(ref, () => {
    console.log("Long pressed");
  });

  return <div ref={ref} />
}
`}
  />
