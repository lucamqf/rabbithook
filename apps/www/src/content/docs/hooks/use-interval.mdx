Tracks the idle state of an user.

### Installation
<Code code="npx rabbithook@latest add use-interval" lang="bash" frame={false} />

### Usage
<Code
    lang="tsx"
    title="App.tsx"
    frame="code"
    code={`
import useInterval from "@/hooks/use-interval";

function Component() {
  useInterval(() => {
    console.log("Executed every 15 seconds");
  }, 15 * 1000 /* 15 seconds */);

  return ();
}
`}
  />
