### Installation
<Code code="npx rabbithook@latest add use-mount" lang="bash" frame={false} />

### Usage
<Code
    lang="tsx"
    title="App.tsx"
    frame="code"
    code={`
import useMount from "@/hooks/use-mount";

function Component() {
  useMount(() => {
    console.log("Component mounted");
  });

  return ();
}
`}
  />
